### 6. 添加切换逻辑 JS（在 `loadAndApplyConfig` 附近）

```js
/* ===== 主题切换 ===== */
var _currentTheme = 'system'; // 'system' | 'light' | 'dark'

function getSystemTheme() {
    return window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
}

function applyTheme(theme) {
    _currentTheme = theme;
    var effective = theme === 'system' ? getSystemTheme() : theme;
    document.documentElement.setAttribute('data-theme', effective);
    // 更新图标
    var icon = document.getElementById('themeIcon');
    if (icon) {
        icon.innerHTML = effective === 'dark'
            ? '<circle cx="12" cy="12" r="5"/><path d="M12 1v2M12 21v2M4.22 4.22l1.42 1.42M18.36 18.36l1.42 1.42M1 12h2M21 12h2M4.22 19.78l1.42-1.42M18.36 5.64l1.42-1.42"/>'
            : '<path d="M21 12.79A9 9 0 1111.21 3a7 7 0 009.79 9.79z"/>';
    }
}

function toggleTheme() {
    // cycle: system → light → dark → system
    var next = _currentTheme === 'system' ? 'light' : _currentTheme === 'light' ? 'dark' : 'system';
    applyTheme(next);
    scheduleConfigSave();
}

// 监听系统主题变化
window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', function() {
    if (_currentTheme === 'system') applyTheme('system');
});
```

### 7. 配置持久化

- `collectConfig()` 中添加 `cfg.theme = _currentTheme`
- `loadAndApplyConfig()` 中添加：
  ```js
  if (cfg.theme) {
      applyTheme(cfg.theme);
  } else {
      applyTheme('system');
  }
  ```
- 在 `loadAndApplyConfig()` 开头（`_loadingConfig = true` 之前）立即调用 `applyTheme('system')` 作为初始默认值，避免白屏闪烁

### 8. 全局操作栏溢出保护

当窗口较窄时，右侧按钮可能被截断。给 `.global-bar` 添加 `overflow:hidden`（已有），确保 `flex:1` spacer 正确压缩左侧内容。

## 验证方式

1. `npm run dev` 启动开发模式
2. 默认状态：跟随系统主题（系统深色 → 深色，系统浅色 → 浅色）
3. 点击太阳图标 → 切换为浅色主题，图标变为月亮
4. 点击月亮图标 → 切换为深色主题，图标变为太阳
5. 再次点击 → 回到系统跟随模式
6. 关闭重启应用，主题应保持上次选择
7. 检查所有 UI 元素（按钮、下拉框、输入框、发送气泡、日志输出、WSL 映射页面）在两种主题下均正常显示
